\documentclass{article}


\usepackage{ifthen}

\def\initializeVariants{
  \newif\ifOne
  \newif\ifTwo
  \Onefalse\Twofalse
  \ifthenelse{\numberOfInstance = 1}{\Onetrue}{}
  \ifthenelse{\numberOfInstance = 2}{\Twotrue}{}
}



\gdef\numberOfInstance{1}
\initializeVariants


\begin{document}

\ifOne This is text only for one\fi

\ifTwo This is text only for two\fi



\end{document}
